  $$

  \begin{enumerate}[label=(\alph*)]
    \item Use Gram-Schmidt procedure to construct an orthonormal basis for $U$.
    \item Find the orthogonal projection of  $\mathbf v = (-1,2,5,1)$ in $U$ and $U^\perp$.
  \end{enumerate}
}
